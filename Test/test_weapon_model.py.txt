import sys
from pathlib import Path
from ultralytics import YOLO

model = YOLO("../models/weapon_detector.pt")

if len(sys.argv) > 1:
    source_path = sys.argv[1]
else:
    # Default to a repository-relative fixture
    source_path = str(Path(__file__).parent / "resources" / "test_image.jpg")

results = model.predict(
    source=source_path,
    conf=0.25,
    show=True
)

for r in results:
    for box in r.boxes:
        x1, y1, x2, y2 = box.xyxy[0].tolist()
        conf = float(box.conf[0])

        print([
            int(x1),
            int(y1),
            int(x2),
            int(y2),
            round(conf, 3)
        ])
